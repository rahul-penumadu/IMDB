SQL> select 
  2  cast(PRIMARYNAME as varchar(40)) as Name, YEAR, COUNT ,case when category = 'actor' and ROW_NUMBER=1 then '1.Actor with most movies'
  3  when category = 'actress' and ROW_NUMBER=1 then '2.Actress with most movies'
  4  when category = 'actor' and ROW_NUMBER>1 then '3.Actor with 3 movies'
  5  when category = 'actress' and ROW_NUMBER>1 then '4.Actress with 4 movies' end as Remark
  6   from (
  7  
  8  SELECT NB.primaryname,min(TB.startyear) year,min(TP.category) category,COUNT(*) count,ROW_NUMBER() OVER (partition by min(TP.category) ORDER BY count(*) desc) AS row_number
  9  FROM imdb00.name_basics NB, imdb00.title_principals TP, imdb00.title_basics TB
 10  WHERE NB.nconst = TP.nconst AND
 11          TP.tconst = TB.tconst AND
 12          LOWER(TB.titleType) = 'movie' AND
 13          LOWER(TP.category) IN ('actor','actress') AND TB.startyear NOT LIKE '\N' and
 14          tb.startyear>='1982' and tb.startyear <='1991'
 15  
 16  GROUP BY NB.nconst, NB.primaryname
 17  ORDER BY COUNT(*) DESC
 18  ) where row_number = '1' or count = '3' order by remark,year
 19  ;
